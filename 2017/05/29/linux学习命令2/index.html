find

find . -name “” -type f -size 0c //查找大小为0文件 find . -name “” -type f -size 0c | xargs -n 1 rm -f //删除空文件 find . -name “*” -type f -size 1024c | xargs -n 1 rm -f //删除指定大小 find -type d -empty //找出空文件夹 find . -name “shuaige.txt” -exec ls {} ; //列出搜索到的文件 find . -name “shuaige.txt” -exec rm -f {} ; //批量删除 find . -name “shuaige.txt” -ok rm -rf {} ; //删除前有提示 find . -name “test” -type d -exec rm -rf {} ; //删除当前目录下面所有 test 文件夹下面的文件 find . -name ‘.svn’ -exec rm -rf {} ; //删除文件夹下面的所有的.svn文件
